## What is Subroute.io?
        
Subroute.io is a hosting and execution environment for C# code built on the .NET framework. Subroute.io is ideal for building microservices,
webhooks, slackbots, or anything that needs to be hosted on the open web. All code written with Subroute.io is compiled as an assembly and
executed in a safe sandbox environment. Your code has full access to the entire .NET framework and has permission to make outgoing web requests.
        
### Writing your First Route
``` csharp
using System;
using System.Net;
using Subroute.Common;
using System.Net.Http;
using System.Linq;
using System.Collections.Generic;
using System.Threading.Tasks;

namespace Subroute.Container
{
    // <summary>
    // Website Monitor
    // Make the request with a "url" query parameter
    // that contains the website to verify uptime.
    // </summary>
    public class RouteMethods : BaseController
    {
	    public async Task<RouteResponse> GetAsync(RouteRequest request)
	    {
		    // Ensure the request has a "url" parameter, otherwise return a 400 bad request.
		    QueryStringValue url;
		    if (!request.QueryString.TryGetValue("url", out url))
		        return Json(HttpStatusCode.BadRequest, new { success = false, message = "You must specify a 'url' query string parameter."});
		    
		    try
		    {
		        // We'll use the HttpClient class to make the outgoing GET request to check
		        // whether the website is up. Subroute now supports await to make the request.
    		    var client = new HttpClient();
    		    var response = await client.GetAsync(url);
		    
		        // Verify that we got a successful response back, then return a 200 OK.
    		    if (response.IsSuccessStatusCode)
    		        return Json(HttpStatusCode.OK, new { success = true, message = "Website is available." });
    		    
    		    // Otherwise we got something other than a success response. So include
    		    // the status code in the response.
		        return Json(HttpStatusCode.OK, new { success = false, message = $"Status code {(int)response.StatusCode} received from server." });
		    }
		    catch (AggregateException ex)
		    {
		        // Sometimes a URL is malformed or some other network exception occurs.
		        // so we capture the exception and return the error message.
		        // AggregateExceptions occur when something fails with an async call.
		        var message = ex?.InnerException?.InnerException.Message;
		        return Json(HttpStatusCode.InternalServerError, new { success = false, message = message });
		    }
		    catch (Exception ex)
		    {
		        // Something else went wrong with the outgoing request. Capture that
		        // and return the error message.
		        var message = ex.Message;
		        return Json(HttpStatusCode.InternalServerError, new { success = false, message = message });
		    }
	    }
    }
}
```
